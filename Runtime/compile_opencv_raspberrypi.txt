OS:
    ubuntu 20.04

# Download Python2.7.18
sudo apt-get update
sudo apt-get install build-essential
sudo apt-get install libssl-dev zlib1g-dev libncurses5-dev libncursesw5-dev libreadline-dev libsqlite3-dev wget curl llvm libncurses5-dev libncursesw5-dev xz-utils tk-dev libffi-dev liblzma-dev python-openssl git

wget https://www.python.org/ftp/python/2.7.18/Python-2.7.18.tgz
tar -xf Python-2.7.18.tgz
cd Python-2.7.18

# Configure and install Python
./configure
make
sudo make install

export PYTHON_LIBRARIES=/usr/local/lib/libpython2.7.so
export PYTHON_INCLUDE_DIRS=/usr/local/include/python2.7


#install Compatible NumPy: If your Python installation is UCS2, you need to install a NumPy version compiled with UCS2 support. pip uninstall numpy
